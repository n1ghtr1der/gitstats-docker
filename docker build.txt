comando exemplo para criação do container com os argumentos necessários

docker build --build-arg <ARG>:<VALUE> . -t <autor>/<image-name>

docker build  --build-arg REPO=git@github.com:n1ghtr1der/booking-app-go.git --build-arg REPO_NAME=booking-app-go --build-arg SSH_PRV_KEY="$(cat ~/.ssh/github)" --build-arg SSH_PUB_KEY="$(cat ~/.ssh/github.pub)" . -t teste/gitstats-image


docker run  --name <container name> -d <image name> tail -f /dev/null

docker build  --build-arg REPO=git@github.com:n1ghtr1der/memoteca-alura.git --build-arg REPO_NAME=memoteca-alura --build-arg SSH_PRV_KEY="$(cat ~/.ssh/github)" . -t teste/gitstats-image
